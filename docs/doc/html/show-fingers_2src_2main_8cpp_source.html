158  /************************************************************************/
159  bool updateModule()
160  {
161  ImageOf<PixelRgb> *imgIn=imgInPort.read(true);
162  if (imgIn==NULL)
163  return false;
164 
165  ImageOf<PixelRgb> &imgOut=imgOutPort.prepare();
166  imgOut=*imgIn;
167 
168  cv::Mat img=cv::cvarrToMat(imgOut.getIplImage());
169 
170  Vector xa,oa;
171  iarm->getPose(xa,oa);
172 
173  Matrix Ha=axis2dcm(oa);
174  xa.push_back(1.0);
175  Ha.setCol(3,xa);
176 
177  Vector pc;
178  igaze->get2DPixel(camSel,xa,pc);
179 
180  cv::Point point_c((int)pc[0],(int)pc[1]);
181  cv::circle(img,point_c,4,cv::Scalar(0,255,0),4);
182 
183  Vector analogs,encs(nEncs),joints;
184  if (ianalog!=NULL)
185  ianalog->read(analogs);
186  iencs->getEncoders(encs.data());
187 
188  for (int i=0; i<3; i++)
189  {
190  if (ianalog!=NULL)
191  finger[i].getChainJoints(encs,analogs,joints);
192  else
193  finger[i].getChainJoints(encs,joints);
194  finger[i].setAng(CTRL_DEG2RAD*joints);
195  }
196 
197  for (int fng=0; fng<3; fng++)
198  {
199  deque<cv::Point> point_f;
200  for (int i=-1; i<(int)finger[fng].getN(); i++)
201  {
202  Vector fc;
203  igaze->get2DPixel(camSel,Ha*(i<0?finger[fng].getH0().getCol(3):
204  finger[fng].getH(i,true).getCol(3)),fc);
205  point_f.push_front(cv::Point((int)fc[0],(int)fc[1]));
206  cv::circle(img,point_f.front(),3,cv::Scalar(0,0,255),4);
207 
208  if (i>=0)
209  {
210  cv::line(img,point_f.front(),point_f.back(),cv::Scalar(255,255,255),2);
211  point_f.pop_back();
212  }
213  else
214  cv::line(img,point_c,point_f.front(),cv::Scalar(255,0,0),2);
215  }
216  }
217 
218  imgOutPort.writeStrict();
219  return true;
220  }
