1 function varargout = specgramscope(varargin)
2 % SPECGRAMSCOPE Live updating spectrogram display
3 %
4 % STEP 1: Initialize the scope
5 % SPECGRAMSCOPE(FS,NFFT) initializes a spectrogram scope in the current axes.
6 % This spectrogram scope will compute and displays the NFFT-point FFT of a vector
7 % signal with sample rate FS Hz. It will show the last 10 FFTs
8 %
9 % STEP 2: Update the scope
10 % SPECGRAMSCOPE(S) updates the spectrogram scope in the current axes with the
11 % FFT of vector S. The scope should first be initialized as above with
12 % sample rate and FFT length. If not, the sample rate will be 1 Hz and the FFT
13 % length will be the length of S. Differences between the length of S and
14 % the specified FFT length are handled the same as MATLAB's built-in FFT
15 % function (i.e., zero-padding or truncation, as appropriate).
16 %
17 % SPECGRAMSCOPE(FS,NFFT,NTRACES) initializes a spectrogram scope in the
18 % current axes with NTRACES traces. This is how many records to keep in
19 % time domain. Default = 10
20 %
21 % SPECGRAMSCOPE(HAX, ...) defines the scope in specified axes HAX instead of GCA. i.e.,
22 % SPECGRAMSCOPE(HAX,FS,NFFT) initializes axes HAX as a spectrogram scope, and
23 % SPECGRAMSCOPE(HAX,S) updates axes HAX with vector S.
24 %
25 % HAX = SPECGRAMSCOPE(...) returns a handle to the axes initialized by the
26 % spectrogram scope. This is useful if you allow SPECGRAMSCOPE to create an
27 % axes for you, and want to be able to easily reference the axes for
28 % updates. The surface created by SPECGRAMSCOPE all have the tag
29 % 'SpecgramScope'. If you would like to manually modify the properties of
30 % these lines, their handles can be found by:
31 %
32 % HAX = SPECGRAMSCOPE(...);
33 % HSurf = findobj(HAX,'Tag','SpecgramScope');
34 %
35 % Example
36 % %% Initialize data
37 % Fs = 16384;
38 % Nfft = 2048;
39 % t = (0:1:Nfft-1)'/Fs;
40 % fo = logspace(3.5,3.7); % Range of fundamental frequencies
41 % s1 = sin(2*pi*t*fo) + .1*rand(Nfft,length(fo));

43 %

253 % Calculate the fft of the data.
254 function [f, mag] = localfft(data,parms)
255 
256 % Calculate the fft of the data.
257 xfft = 2/parms.Nfft*fft(data,parms.Nfft);
258 
259 % Avoid taking the log of 0.
260 xfft(xfft == 0) = 1e-17;
261 
262 % Compute magnitude, dB
263 mag = 20*log10(abs(xfft(1:parms.Nfft/2,:)));
264 
265 f = (0:length(mag)-1)*parms.Fs/parms.Nfft;
266 f = f(:);
